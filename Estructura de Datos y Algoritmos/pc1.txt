public class pc1 {

    public static int parteA(int numero, int digito){
        int contador = 0;
        int residuo;
        int cociente;

        while (numero > 0){
            residuo = numero % 10;
            cociente = numero / 10;

            if (residuo == digito) contador++;

            numero = cociente;
        }

        return contador;
    }

    public static boolean parteB(int numero){
        int digito4 = parteA(numero, 4);
        int digito7 = parteA(numero, 7);

        if (digito4 >= 3 || digito7 >= 2)
            return true;
        else
            return false;
    }

    public static void pregunta1(){
        int cantidad = parteA(4351467, 4);
        System.out.println(cantidad);
        boolean resultado = parteB(4351467);
        System.out.println(resultado);
    }

    public static int parteC(double[] montoVenta){
        int contador = 0;

        for (int i=0; i<montoVenta.length; i++){
            if (montoVenta[i]<400)
                contador += 1;
        }
        return contador;
    }

    public static int parteD(String[] codCliente, String cliente){
        int contador = 0;
        for (int i=0; i<codCliente.length; i++){
            if (codCliente[i].equals(cliente))
                contador ++;
        }
        return contador;
    }
